Backbone.LazyView Extended Backbone.View for lazy loading

Fork me on GitHub.

Demo

var LazyView = Backbone.LazyView.extend({
    events: {
        appear: 'appearHandler'
    },
    appearHandler: function () {
        $('#output').append('<br/>' + this.options.name + ' appears');
    }
});
new LazyView({ el: '#view1', name: 'view1' });
new LazyView({ el: '#view2', name: 'view2', threshold: 50 });
new LazyView({ el: '#view3', name: 'view3', threshold: -100 });
